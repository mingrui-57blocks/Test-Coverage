import moment from "moment-timezone";
 
function toLocalISOString(date, is_replace_T = true, is_trim_millisecond = true) {
  let offset = date.getTimezoneOffset();
  let tzoffset = offset * 60000;
  let localISOTimestr = new Date(date - tzoffset).toISOString();
  if (is_replace_T) {
    localISOTimestr = localISOTimestr.replace("T", " ");
  }
  if (is_trim_millisecond) {
    localISOTimestr = localISOTimestr.split(".")[0];
  }
  return localISOTimestr;
}
 
function toLocalDateString(date) {
  return toLocalISOString(date, false, false).split("T")[0];
}
 
function toLocalString(date, is_replace_T = true, is_trim_millisecond = true) {
  return toLocalISOString(date, is_replace_T, is_trim_millisecond);
}
 
function toTimeZoneDateString(date, timezone) {
  return toTimeZoneString(date, timezone).split(" ")[0];
}
 
function toTimeZoneString(date, timezone, is_trim_millisecond = true) {
  if (!timezone || timezone === "" || timezone === "local" || timezone === "Local") {
    return toLocalISOString(date, true, is_trim_millisecond).replace("Z", "");
  } else {
    if (is_trim_millisecond) {
      return moment(date)
        .tz(timezone)
        .format("YYYY-MM-DD HH:mm:ss");
    } else {
      return moment(date)
        .tz(timezone)
        .format("YYYY-MM-DD HH:mm:ss.SSS");
    }
  }
}
 
function unix_time_from_datestr(datestr, timezone) {
  if (!timezone || timezone === "" || timezone === "local" || timezone === "Local") {
    let tmp_date = new Date();
    let keyarr = datestr.split("-");
    tmp_date.setHours(0, 0, 0, 0);
    tmp_date.setFullYear(keyarr[0], +keyarr[1] - 1, keyarr[2]);
    return tmp_date.getTime();
  } else {
    return moment.tz(datestr, timezone).valueOf();
  }
}
 
function round_time_to_resolution(time, resolution, rewind = true) {
  let tmp_time = new Date(time);
  if (resolution === "1h" || resolution === "1d") {
    if (rewind) {
      tmp_time.setMinutes(0, 0, 0);
    } else if (tmp_time.getMinutes() !== 0) {
      tmp_time.setMinutes(0, 0, 0);
      tmp_time.setHours(tmp_time.getHours() + 1);
    }
  } else if (resolution === "15m") {
    let minutes = tmp_time.getMinutes();
    if (rewind) {
      tmp_time.setMinutes(minutes - (minutes % 15), 0, 0);
    } else if (minutes % 15 !== 0) {
      tmp_time.setMinutes(minutes + (15 - (minutes % 15)), 0, 0);
    }
  }
  tmp_time.setSeconds(0, 0);

  return tmp_time;
}
 
function get_start_end_time_days(days, timezone) {
  let today = new Date();
  today.setDate(today.getDate());
  today.setHours(0, 0, 0, 0);

  if (timezone && timezone !== "local") {
    today = new Date(
      moment(new Date())
        .tz(timezone)
        .set({ hour: 0, minute: 0, second: 0, millisecond: 0 })
        .format()
    );
  }

  let startday, endday;
  if (days >= 0) {
    let calc_date = new Date(today);
    calc_date.setDate(calc_date.getDate() + days + 1);

    startday = today;
    endday = calc_date;
  } else {
    let calc_date = new Date(today);
    calc_date.setDate(calc_date.getDate() + days);

    startday = calc_date;
    endday = today;
  }

  return {
    start: startday.toISOString(),
    end: endday.toISOString()
  };
}
 
function get_start_end_time_weeks(weeks, timezone) {
  let last_sunday = new Date();
  last_sunday.setHours(0, 0, 0, 0);
  last_sunday.setDate(last_sunday.getDate() - last_sunday.getDay());

  if (timezone && timezone !== "local") {
    last_sunday = new Date(
      moment(new Date())
        .tz(timezone)
        .set({ hour: 0, minute: 0, second: 0, millisecond: 0 })
        .format()
    );

    var day = new Date().toLocaleString("en-US", {
      timeZone: timezone
    });
    day = new Date(day);
    day.getDay();

    last_sunday.setDate(last_sunday.getDate() - day.getDay());
  }

  let startday, endday;
  if (weeks >= 0) {
    let calc_date = new Date(last_sunday);
    calc_date.setDate(calc_date.getDate() + 7 * (weeks + 1));

    startday = last_sunday;
    endday = calc_date;
  } else {
    let calc_date = new Date(last_sunday);
    calc_date.setDate(calc_date.getDate() + 7 * weeks);

    startday = calc_date;
    endday = last_sunday;
  }

  return {
    start: startday.toISOString(),
    end: endday.toISOString()
  };
}
 
function utc_to_timezone_str(input, timezone) {
  let returnstr = "";
  if (timezone === "" || timezone === "Local" || timezone === "local") {
    returnstr = new Date(input.replace("Z", "")).toISOString();
  } else {
    returnstr = moment.tz(input.replace("Z", ""), timezone).toISOString();
  }
 
  return returnstr;
}
