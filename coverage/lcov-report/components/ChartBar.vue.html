<template>
  <div id="chartbar" :class="{ processing: processing }">
    <ProcessingVue :processing="processing" />
    <div id="dummy">
      <svg :id="get_dummyTextid" />
    </div>
    <div v-show="data.length === 0 && !processing" id="nodatatext">
      {{ $t("data.noAvailableData") }}
    </div>
    <div v-show="data.length > 0 && !processing">
      <svg :id="get_SVGid" />
    </div>
  </div>
</template>
 
<script>
import * as d3 from "d3";
import _ from "lodash";
import ProcessingVue from "@/components/Processing.vue";
import mixins_uuid from "@/mixins/mixins_uuid.js";
 
export default {
  components: { ProcessingVue },
  mixins: [mixins_uuid],
  props: {
    data: { type: Array, default: () => [] },
    uid_key: { type: String, default: "id" },
    y_label: { type: String, default: "" },
    x_label: { type: String, default: "" },
    y_max: { type: Number, default: 1 },
    sorting: { type: String, default: "" },
    unit: { type: String, default: "" },
    id_to_color: { type: Object, default: () => {} },
    processing: { type: Boolean, default: false },
    si_format: { type: Boolean, default: true }
  },
  data() {
    return {
      chart: null
    };
  },
  computed: {
    get_SVGid() {
      return "svg_" + this.uuid;
    },
    get_dummyTextid() {
      return "dummytext_" + this.uuid;
    },
    get_dummyTextgclass() {
      return "dummytextg_" + this.uuid;
    }
  },
  watch: {
    data(val) {
      this.renderChart(val);
    },
    y_label() {
      this.renderChart(this.data);
    },
    x_label() {
      this.renderChart(this.data);
    },
    unit() {
      this.renderChart(this.data);
    },
    sorting() {
      this.renderChart(this.data);
    }
  },
  mounted() {
    this.handleResize();
    window.addEventListener("resize", this.handleResize);
  },
  beforeDestroy() {
    // Unregister the event listener before destroying this Vue instance
    window.removeEventListener("resize", this.handleResize);
  },
  methods: {
    handleResize() {
      this.renderChart(this.data);
    },
    renderChart(input_data) {
      if (this.chart != null) this.chart.remove();
      if (!input_data || input_data.length <= 0) return;
      const margin_top = 32;
      const margin_bottom = 64;
      const margin_right = 64;
      const margin_left = 64;
      const svg_width = this.$el.clientWidth;
      const svg_height = this.$el.clientHeight;
      const chart_width = svg_width - margin_right - margin_left;
      const chart_height = svg_height - margin_top - margin_bottom;
      const unit = this.unit;
      const uid_key = this.uid_key;
      let numFormat = this.si_format ? d3.format(".3~s") : d3.format(".3f");
      let original_data = JSON.parse(JSON.stringify(input_data));
 
      if (this.sorting === "ASC") {
        input_data = original_data.sort(function(a, b) {
          return d3.ascending(a.value, b.value);
        });
      } else if (this.sorting === "DESC") {
        input_data = original_data.sort(function(a, b) {
          return d3.descending(a.value, b.value);
        });
      } else {
        input_data = original_data;
      }
      // svg
      const svg = d3
        .select("#" + this.get_SVGid)
        .attr("width", svg_width)
        .attr("height", svg_height);
      this.chart = svg.append("g").attr("transform", `translate(${margin_left}, ${margin_top})`);
 
      // define yscale
      let yValMax = _.maxBy(input_data, "value").value;
      if (yValMax < this.y_max) yValMax = this.y_max;
      let yValMin = _.minBy(input_data, "value").value;
      if (yValMin > 0) yValMin = 0;
      const yScale = d3
        .scaleLinear()
        .range([chart_height, 0])
        .domain([yValMin, yValMax]);
 
      // define xscale
      const xScale = d3
        .scaleBand()
        .range([0, chart_width])
        .domain(input_data.map(s => s[uid_key]))
        .padding(0.5);
 
      // define color schema
      let color = "";
      if (this.id_to_color != null) {
        color = d3
          .scaleOrdinal()
          .domain(Object.keys(this.id_to_color))
          .range(Object.values(this.id_to_color));
      } else {
        // list of group names
        let res = input_data.map(function(d) {
          return d.name;
        });
        color = d3
          .scaleOrdinal()
          .domain(res)
          .range(["#0fc8c3"]);
      }
 
      // append background box to catch mouse motion within a wider range for tooltips
      const bgBox = this.chart
        .append("g")
        .attr("clip-path", "url(#clip)")
        .append("rect")
        .attr("x", 1) // so that y-axis would not be covered
        .attr("y", 0)
        .attr("width", chart_width - 1)
        .attr("height", chart_height)
        .attr("fill", "#ffffff")
        .attr("opacity", 0.0);
 
      // bar group
      const barGroups = this.chart
        .selectAll()
        .data(input_data)
        .enter()
        .append("g");
 
      barGroups
        .append("rect")
        .attr("class", "bar")
        .attr("fill", function(d) {
          return color(d.id);
        })
        .attr("x", g => xScale(g[uid_key]))
        .attr("y", g => (g.value >= 0 ? yScale(g.value) : yScale(0)))
        .attr("height", g => Math.abs(yScale(0) - yScale(g.value)))
        .attr("width", xScale.bandwidth())
        .append("svg:title")
        .text(function(d) {
          return d.name;
        });
 
      // draw yaxis
      this.chart
        .append("g")
        .call(
          d3
            .axisLeft(yScale)
            .ticks(7)
            .tickFormat(numFormat)
        )
        .attr("color", "#657c77");
 
      // draw xaxis
      let tickPaddingVal = chart_height - yScale(0) + 10;
      let tickSizeVal = 0;
      this.chart
        .append("g")
        .attr("class", "xAxis")
        .attr("transform", `translate(0, ${yScale(0)})`)
        .call(
          d3
            .axisBottom(xScale)
            .tickSize(tickSizeVal)
            .tickPadding(tickPaddingVal)
            .tickFormat(function(d) {
              const bar = input_data.filter(b => b[uid_key] === d)[0];
              if (bar) return bar.name;
              else return d;
            })
        )
        .attr("color", "#657c77");
 
      // rotate x-tick labels if overlapping
      let xTextWidth = this.get_xText_width(input_data);
      let maxXTextWidth = Math.max(...xTextWidth);
      // factor 1.05 to ensure some space between labels
      if (maxXTextWidth > xScale.step() / 1.05) {
        let angle = (Math.acos(xScale.step() / maxXTextWidth / 1.05) * 180) / Math.PI;
        this.chart
          .select(".xAxis")
          .selectAll("text")
          .style("text-anchor", "start")
          .attr("transform", "rotate(" + angle + ",0," + (tickPaddingVal + tickSizeVal) + ")");
      }
 
      // add axis labels
      let unitText = "";
      if (unit.length > 0) {
        unitText = "(" + unit + ")";
      }
      if (this.y_label) {
        this.chart
          .append("text")
          .attr("transform", "rotate(-90)")
          .attr("y", -45)
          .attr("x", -chart_height / 2)
          .style("font-size", 12)
          .attr("color", "#657c77")
          .style("text-anchor", "middle")
          .text(this.y_label + " " + unitText);
      }
      if (this.x_label) {
        this.chart
          .append("text")
          .attr("y", chart_height)
          .attr("dy", chart_height * 0.12)
          .attr("x", chart_width / 2)
          .attr("color", "#657c77")
          .style("text-anchor", "middle")
          .style("font-size", 12)
          .text(this.x_label);
      }
 
      // add tooltips
      this.chart
        .on("mousemove", addTooltip)
        .on("mouseleave", removeTooltip)
        .style("pointer-events", "all");
 
      let tooltips = this.chart.append("g").attr("class", "valTooltip");
 
      tooltips
        .append("text")
        .attr("class", "val")
        .attr("x", 0)
        .attr("y", 0)
        .style("font-size", 10)
        .style("text-anchor", "middle");
 
      let barPosObj = input_data.map(function(d) {
        let tmpObj = {};
        tmpObj[uid_key] = d[uid_key];
        tmpObj["barPos"] = xScale(d[uid_key]) + xScale.bandwidth() / 2;
        return tmpObj;
      });
 
      function addTooltip() {
        if (
          d3.mouse(this)[0] >= 0 &&
          d3.mouse(this)[0] <= chart_width &&
          d3.mouse(this)[1] >= 0 &&
          d3.mouse(this)[1] <= chart_height
        ) {
          tooltips.style("display", null);
          barGroups.selectAll("rect").attr("opacity", 0.3);
 
          let xMouse = d3.mouse(this)[0];
          let xNearest = chart_width;
          let idNearest = -1;
          barPosObj.forEach(function(d) {
            let distToD = Math.abs(xMouse - d["barPos"]);
            if (distToD < xNearest) {
              xNearest = distToD;
              idNearest = d[uid_key];
            }
          });
 
          let targetedBar = barGroups.selectAll("rect").filter(function(d) {
            return d[uid_key] === idNearest;
          });
          targetedBar.attr("opacity", 1.0);
 
          let targetedData = input_data.filter(d => d[uid_key] === idNearest)[0];
          let valStr = targetedData.value.toFixed(0).toString();
          let xPos = xScale(idNearest) + xScale.bandwidth() / 2;
          let yPos = targetedData.value >= 0 ? yScale(targetedData.value) - 5 : yScale(0) - 5;
          tooltips
            .select(".val")
            .text(valStr)
            .attr("x", xPos)
            .attr("y", yPos)
            .attr("fill", color(idNearest));
        } else {
          barGroups.selectAll("rect").attr("opacity", 1.0);
        }
      }
      function removeTooltip() {
        barGroups.selectAll("rect").attr("opacity", 1.0);
 
        tooltips.selectAll("line").remove();
        tooltips.style("display", "none");
      }
    },
    get_xText_width(input_data) {
      // pre-calculate the space needed for the longest name
      var descTextArray = input_data.map(d => d.name);
      var textWidthArr = [];
      const svg = d3.select("#" + this.get_dummyTextid);
 
      svg
        .append("g")
        .attr("class", this.get_dummyTextgclass)
        .selectAll("." + this.get_dummyTextgclass)
        .data(descTextArray)
        .enter()
        .append("text")
        .attr("font-family", "sans-serif")
        .attr("font-size", "10")
        .attr("opacity", 0.0)
        .text(function(d) {
          return d;
        })
        .each(function(d, i) {
          var thisWidth = this.getComputedTextLength();
          textWidthArr.push(thisWidth);
        });
      svg.selectAll("." + this.get_dummyTextgclass).remove();
      return textWidthArr;
    }
  }
};
</script>
